Track how buyers find and engage with your listings using traffic analytics.

## Getting Traffic Data

```javascript
// Get listing traffic for the past 30 days
const report = await mcp.useTool('analytics_getTrafficReport', {
  dimension: 'LISTING',
  filter: 'listing_ids:{1234567890|9876543210}',
  metric_keys: ['CLICK_THROUGH_RATE,LISTING_IMPRESSION,LISTING_VIEWS,TRANSACTION'],
  start_date: '2024-01-01',
  end_date: '2024-01-31'
});

report.records.forEach(record => {
  console.log(`Listing: ${record.listing_id}`);
  console.log(`  Impressions: ${record.listing_impression}`);
  console.log(`  Views: ${record.listing_views}`);
  console.log(`  CTR: ${record.click_through_rate}%`);
  console.log(`  Sales: ${record.transaction}`);
  console.log(`  Conversion: ${(record.transaction / record.listing_views * 100).toFixed(2)}%`);
});
```

## Available Dimensions

| Dimension | Description |
|-----------|-------------|
| `LISTING` | Per-listing metrics |
| `DAY` | Daily aggregated metrics |
| `TRAFFIC_SOURCE` | Metrics by traffic source (search, browse, etc.) |

## Example: Weekly Performance Report

```javascript
async function weeklyPerformanceReport() {
  const endDate = new Date();
  const startDate = new Date(endDate.getTime() - 7 * 24 * 60 * 60 * 1000);

  const report = await mcp.useTool('analytics_getTrafficReport', {
    dimension: 'DAY',
    metric_keys: ['LISTING_IMPRESSION,LISTING_VIEWS,TRANSACTION'],
    start_date: startDate.toISOString().split('T')[0],
    end_date: endDate.toISOString().split('T')[0]
  });

  console.log('ðŸ“Š Weekly Performance Report\n');

  let totalImpressions = 0;
  let totalViews = 0;
  let totalSales = 0;

  report.records.forEach(record => {
    totalImpressions += record.listing_impression || 0;
    totalViews += record.listing_views || 0;
    totalSales += record.transaction || 0;

    console.log(`${record.date}:`);
    console.log(`  Impressions: ${record.listing_impression}`);
    console.log(`  Views: ${record.listing_views}`);
    console.log(`  Sales: ${record.transaction}`);
  });

  console.log('\nðŸ“ˆ Weekly Totals:');
  console.log(`Total Impressions: ${totalImpressions}`);
  console.log(`Total Views: ${totalViews}`);
  console.log(`Total Sales: ${totalSales}`);
  console.log(`Average CTR: ${(totalViews / totalImpressions * 100).toFixed(2)}%`);
  console.log(`Average Conversion: ${(totalSales / totalViews * 100).toFixed(2)}%`);
}

await weeklyPerformanceReport();
```
